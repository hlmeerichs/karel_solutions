// second, clean solution
void tileTheFloor()
{
    dropBeeper();
    repeat (118)
    {
        if (! beeperAhead())
        {
            if (frontIsClear())
            {
                moveForward();
                dropBeeper();
            }
            else
            {
                turnLeft();
            }
        }
        
        else
        {
            turnLeft();
        }
    }
}
